export VAR!@#=test    # Δεν υποστηρίζονται ειδικοί χαρακτήρες στο όνομα
export VAR VALUE      # Λείπει το =
export "VAR WITH SPACE"=test  # Το όνομα της μεταβλητής δεν μπορεί να έχει κενά
export VAR+=test      # Το += δεν υποστηρίζεται για μεταβλητές περιβάλλοντος
export VAR==value     # Διπλά ==
export export VAR=value # Δεν χρειάζεται διπλό export
export VAR=value; export VAR2 # Η δεύτερη εντολή είναι ελλιπής
export VAR=$UNDEFINED_VAR # Αν η $UNDEFINED_VAR είναι άδεια, το VAR θα είναι κενό
export 1VAR=test      # Το όνομα της μεταβλητής δεν μπορεί να ξεκινά με αριθμό
export VAR-TEST=foo   # Δεν επιτρέπονται παύλες στο όνομα της μεταβλητής
export VAR&TEST=foo   # Το `&` δεν επιτρέπεται στο όνομα
export $VAR=value     # Δεν μπορεί να ξεκινά με `$`
export VAR=test test2 # Το test2 θεωρείται ξεχωριστό όρισμα
export VAR==          # Λάθος χρήση του `=`
export VAR="unterminated # Λάθος, λείπει το τελικό `"`
export PATH+=":/custom/bin"  # Δεν υποστηρίζεται += στο minishell
export VAR=test; ls          # `ls` δεν πρέπει να είναι στην ίδια γραμμή
export -n VAR=test           # Το `-n` δεν είναι έγκυρο flag στο minishell
export -p VAR=test           # To `-p` στο minishell δεν σημαίνει κάτι

export VAR="$(command notfound)"  # Αν η εντολή αποτύχει, η μεταβλητή θα είναι κενή
export VAR=$'test\n'         # Το `$'...'` ίσως να μην υποστηρίζεται στο minishell
export VAR=test | grep VAR   # Η εντολή `grep` δεν ανήκει σε `export`
