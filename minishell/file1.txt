echo nice > file1 > file2 < grep "nice" < ls | cat

echo nice > file1 :
	entree : stdin
	sortie : file1
	-> file 1 : "nice"

file1 > file2 :
	entree : file1
	sortie : file2
	-> file 2 : "nice"

file2 < grep "nice" :
	entree : file2
	sortie : stdout
	-> stdout : 1

grep "nice" < ls :
	entree : stdin
	sortie : stdout
	-> stdout : (ex : 1 car 1 fichier, file1)

ls | cat :
	entree : stdin
	sortie : stdout
	-> stdout : 1
